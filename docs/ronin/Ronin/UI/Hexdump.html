def Hexdump.dump(object,output=STDOUT)
  unless object.respond_to?(:each_byte)
    raise(ArgumentError,"argument must respond to #each_byte")
  end

  index = 0
  offset = 0
  hex_segment = nil
  print_segment = nil

  segment = lambda {
    output.printf(
      "%.8x  %s  |%s|\n",
      index,
      hex_segment.join(' ').ljust(47).insert(23,' '),
      print_segment.join
    )
  }

  object.each_byte do |b|
    if offset == 0
      hex_segment = []
      print_segment = []
    end

    hex_segment << ("%.2x" % b)

    print_segment[offset] = case b
                            when (0x20..0x7e)
                              b.chr
                            else
                              '.'
                            end

    offset += 1

    if (offset >= 16)
      segment.call

      offset = 0
      index += 16
    end
  end

  unless offset == 0
    segment.call
  end

  return nil
end
