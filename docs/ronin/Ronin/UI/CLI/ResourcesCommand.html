def print_resources(resources)
  if options.csv?
    print resources.to_csv
  elsif options.xml?
    print resources.to_xml
  elsif options.yaml?
    print resources.to_yaml
  elsif options.json?
    print resources.to_json
  else
    resources.each { |resource| print_resource(resource) }
  end
end
