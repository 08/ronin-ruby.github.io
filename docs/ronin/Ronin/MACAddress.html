def recent_ip_address
  self.ip_address_mac_addresses.all(
    :order => [:created_at.desc]
  ).ip_addresses.first
end

def to_i
  self.address.split(':').inject(0) do |bits,char|
    bits = ((bits << 8) | char.hex)
  end
end
