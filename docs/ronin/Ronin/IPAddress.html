def self.lookup(name)
  host = HostName.first_or_new(:address => name)
  ips = begin
          Resolv.getaddresses(name)
        rescue
          []
        end
    
  ips.map! do |addr|
    IPAddress.first_or_create(
      :address => addr,
      :host_names => [host]
    )
  end

  return ips
end

def lookup!
  hosts = begin
            Resolv.getnames(self.address.to_s)
          rescue
            []
          end

  hosts.map! do |name|
    HostName.first_or_create(
      :address => name,
      :ip_addresses => [self]
    )
  end

  return hosts
end
