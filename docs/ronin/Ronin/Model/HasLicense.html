def self.included(base)
  base.send :include, Model
  base.send :extend, ClassMethods

  base.module_eval do
    # The license
    belongs_to :license, Ronin::License, :required => false

    Ronin::License.has 0..n, self.relationship_name, :model => self
  end
end
