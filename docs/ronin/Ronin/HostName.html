def self.lookup(addr)
  addr = addr.to_s
  ip = IPAddress.first_or_new(:address => addr)
  hosts = begin
            Resolv.getnames(addr)
          rescue
            []
          end

  hosts.map! do |name|
    HostName.first_or_create(
      :address => name,
      :ip_addresses => [ip]
    )
  end

  return hosts
end

def lookup!
  ips = begin
          Resolv.getaddresses(self.address)
        rescue
          []
        end

  ips.map! do |addr|
    IPAddress.first_or_create(
      :address => addr,
      :host_names => [self]
    )
  end

  return ips
end
