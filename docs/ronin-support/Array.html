Examples:

# => [0x41, 0x41, 0x20]

def bytes
  self.inject([]) do |accum,elem|
    if elem.kind_of?(Integer)
      accum << elem
    else
      elem.to_s.each_byte { |b| accum << b }
    end

    accum
  end
end

def chars
  array_bytes = self.bytes

  array_bytes.map! { |b| b.chr }
  return array_bytes
end

def hex_chars
  array_bytes = self.bytes

  return array_bytes
end
