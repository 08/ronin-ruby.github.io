def SSL.verify(mode=nil)
  verify_mode = 'VERIFY_' + (mode || :none).to_s.upcase

  unless OpenSSL::SSL.const_defined?(verify_mode)
    raise(RuntimeError,"unknown verify mode #{mode}")
  end

  return OpenSSL::SSL.const_get(verify_mode)
end
